## Basic command line interface navigation

pwd - print working directory
ls - list files in directory
cd - move to a file
directory - folder
mv *filename* *newname* - change file name
cp -r ORIGINAL COPY - copy original to copy
rmdir EMPTYDIR - removes empty directory
rm filename - removes file 
rm -r folder - removes folder and its contents
ssh tero@example.com - open remote command shell
remotecomputer$ exit - exit machine
scp -r FOLDER tero@example.com:public_html/ - securely copy a folder to a folder in remote machine
man ls - manual page of command ( " " for next page, "q" to quit, "b" to move backwards, "/" searches

## Directories

/ - root directory
/home/ - contains home directories for all users
/home/*username*/ - home directory of user, 
/etc/ - all system wide settings
/media/ - removable media

## installing software

sudo apt-get update - update available packages
apt-cache search dungeon adventure - search for software
sudo apt-get -y install nethack-console - install software
